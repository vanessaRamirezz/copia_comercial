// Función para generar datos aleatorios
function llenarFormularioReferenciasNoFamiliares() {
    const nombres = ["Carlos García", "Ana López", "Juan Pérez", "Marta Rodríguez"];
    const direcciones = ["Av. Libertad 234, San Salvador", "Calle 15, Santa Ana", "Calle El Sol, Usulután", "Boulevard de la Reforma"];
    const telefonos = () => `${Math.floor(Math.random() * 9000 + 1000)}-${Math.floor(Math.random() * 9000 + 1000)}`;
    const lugaresTrabajo = ["Comercial de Servicios S.A.", "Servicios Financieros S.A.", "Corporación X", "Consultora ABC"];

    // Generación aleatoria de datos
    const nombreUno = nombres[Math.floor(Math.random() * nombres.length)];
    const direccionUno = direcciones[Math.floor(Math.random() * direcciones.length)];
    const telefonoUno = telefonos();
    const lugarTrabajoUno = lugaresTrabajo[Math.floor(Math.random() * lugaresTrabajo.length)];
    const telefonoTrabajoUno = telefonos();

    const nombreDos = nombres[Math.floor(Math.random() * nombres.length)];
    const direccionDos = direcciones[Math.floor(Math.random() * direcciones.length)];
    const telefonoDos = telefonos();
    const lugarTrabajoDos = lugaresTrabajo[Math.floor(Math.random() * lugaresTrabajo.length)];
    const telefonoTrabajoDos = telefonos();

    // Llenar el formulario
    document.getElementById('nombreRefNoFamiliarUno').value = nombreUno;
    document.getElementById('dirRefNoFamiliarUno').value = direccionUno;
    document.getElementById('telProRefNoFamiliarUno').value = telefonoUno;
    document.getElementById('lugarTrabajoRefNoFamiliarUno').value = lugarTrabajoUno;
    document.getElementById('telTrabaRefNoFamiliarUno').value = telefonoTrabajoUno;

    document.getElementById('nombreRefNoFamiliarDos').value = nombreDos;
    document.getElementById('dirRefNoFamiliarDos').value = direccionDos;
    document.getElementById('telProRefNoFamiliarDos').value = telefonoDos;
    document.getElementById('lugarTrabajoRefNoFamiliarDos').value = lugarTrabajoDos;
    document.getElementById('telTrabaRefNoFamiliarDos').value = telefonoTrabajoDos;
}

// Llamar a la función para llenar los datos cuando sea necesario
llenarFormularioReferenciasNoFamiliares();
